The following code clips an entire group of raster files with a designated shapefile:

#Create a new script in a toolbox and run this code:
poly = arcpy.GetParameterAsText(0) #Create shapefile input variable for clip.
folder = arcpy.GetParameterAsText(1) #Create folder input variable to clip.
arcpy.env.workspace = folder

rasterlist = arcpy.ListRasters()
for r in rasterlist:
arcpy.AddMessage(r) #prints names of rasters, NOT NEEDED
newname = r + "_Clip.tif" #outputs in same folder with _Clip.tif added.
arcpy.Clip_management(r,"#",newname,poly,0, "ClippingGeometry")
